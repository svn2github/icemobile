The ViewManager component provides an easy way to build mobile-friendly, responsive UIs. The component can automatically build a multi-layered menu structure. 

    <mobi:viewManager title="ICEmobile" 
        transitionType="#{viewManagerBean.transitionType}"
        headerStyle="#{viewManagerBean.headerStyle}"
        selected="#{viewManagerBean.view}"
        history="#{viewManagerBean.history}">
        <mobi:view id="viewmanager" title="View Manager" menuIcon="th-list"
           navBarGroup="viewmanager" divider="Layout and Navigation">
            ...content
        </mobi:view>
        <mobi:view id="viewmanagerDocs" title="Docs" menuIcon="book"
            includeInMenu="false" navBarGroup="viewmanager">
           ...content
        </mobi:view>
    </mobi:viewManager>
