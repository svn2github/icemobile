The carousel component displays a list of child elements, allowing a user to swipe right or left. The most common use of this component is for displaying a list of images that the user can swipe to view additional images in the list. 

<mobi:carousel 
  selectedItem="#{bean.selectedItem}"
  value="#{bean.images}"
  singleSubmit="true"
  var="item">
  <h:panelGroup>
    <h:graphicImage library="images" 
      name="#{item}"/>
  </h:panelGroup>
</mobi:carousel>
