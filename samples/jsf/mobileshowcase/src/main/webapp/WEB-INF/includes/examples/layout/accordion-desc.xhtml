The accordion component contains child contentPane components. Only one content pane can be expanded at any given time. The title bar text for each accordion panel can be set via the title attribute on the contentPane.

<mobi:accordion 
  selectedId="#{bean.selectedId}"
  autoHeight="true">
  <mobi:contentPane title="Tab 1" 
    client="true">
    ...
  </mobi:contentPane>
  <mobi:contentPane title="Tab 2" 
    client="true">
    ...
  </mobi:contentPane>
  <mobi:contentPane title="Tab 3" 
    client="true">
    ...
  </mobi:contentPane>
</mobi:accordion>
