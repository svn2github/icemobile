The accordion component contains child contentPane components. Only one content pane can be expanded at any given time. The title bar text for each accordion panel can be set via the title attribute on the contentPane.

<mobi:accordion id="autoHeightAccordion" selectedId="⌗{accordionBean.selectedId}"
    autoHeight="true">
    <mobi:contentPane title="ICE Sailor" id="accordionPane1" client="true">
            <ui:include src="/WEB-INF/includes/content/inc-icesailor.xhtml"/>
    </mobi:contentPane>
    <mobi:contentPane title="ICE Breaker" id="accordionPane2" client="true">
        <ui:include src="/WEB-INF/includes/content/inc-icebreaker.xhtml"/>
    </mobi:contentPane>
    <mobi:contentPane title="ICE Skate" id="accordionPane3" client="true">
        <ui:include src="/WEB-INF/includes/content/inc-iceskater.xhtml"/>
    </mobi:contentPane>
</mobi:accordion>
